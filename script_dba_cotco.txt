CREATE SCHEMA `schemapadrao` ;
use `schemapadrao` ;
CREATE TABLE TB_EMPRESA (
    ID_EMPRESA BIGINT AUTO_INCREMENT PRIMARY KEY,
    RESP_EMPRESA VARCHAR(255) NOT NULL,
    RZSOCIAL_EMPRESA VARCHAR(255) NOT NULL,
    NMFANT_EMPRESA VARCHAR(255),
    CNPJ_EMPRESA VARCHAR(255) NOT NULL,
    TEL_EMPRESA VARCHAR(255) NOT NULL,
    EMAIL_EMPRESA VARCHAR(255) NOT NULL,
    SITUACAO_EMPRESA VARCHAR(255),
    SENHA_EMPRESA VARCHAR(8) NOT NULL
);

select * from TB_EMPRESA
